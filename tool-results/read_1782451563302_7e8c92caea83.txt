   133→function formatRelativeTime(dateStr: string): string {
   134→  const date = new Date(dateStr);
   135→  const now = new Date();
   136→  const diffMs = now.getTime() - date.getTime();
   137→  const diffSec = Math.floor(diffMs / 1000);
   138→  const diffMin = Math.floor(diffSec / 60);
   139→  const diffHr = Math.floor(diffMin / 60);
   140→  const diffDay = Math.floor(diffHr / 24);
   141→
   142→  if (diffSec < 60) return 'just now';
   143→  if (diffMin < 60) return `${diffMin}m ago`;
   144→  if (diffHr < 24) return `${diffHr}h ago`;
   145→  if (diffDay < 7) return `${diffDay}d ago`;
   146→  if (diffDay < 30) return `${Math.floor(diffDay / 7)}w ago`;
   147→  return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
   148→}
