   243→// ── Helpers ─────────────────────────────────────────────────────────────
   244→
   245→function formatTimeAgo(date: Date): string {
   246→  const seconds = Math.floor((Date.now() - date.getTime()) / 1000);
   247→  if (seconds < 60) return `${seconds}s ago`;
   248→  const minutes = Math.floor(seconds / 60);
   249→  if (minutes < 60) return `${minutes} min ago`;
   250→  const hours = Math.floor(minutes / 60);
   251→  if (hours < 24) return `${hours} hour${hours > 1 ? 's' : ''} ago`;
   252→  const days = Math.floor(hours / 24);
   253→  return `${days} day${days > 1 ? 's' : ''} ago`;
   254→}
   255→
   256→function isToday(date: Date): boolean {
   257→  const now = new Date();
   258→  return date.toDateString() === now.toDateString();
   259→}
   260→
   261→function isWithinDays(date: Date, days: number): boolean {
   262→  const cutoff = new Date(Date.now() - days * 24 * 60 * 60 * 1000);
   263→  return date >= cutoff;
   264→}
