   232→  const [newPersonaName, setNewPersonaName] = useState('');
   233→  const [newPersonaPrompt, setNewPersonaPrompt] = useState('');
   234→
   235→  const [rules, setRules] = useState<Rule[]>(PRESET_RULES);
   236→  const [newRuleName, setNewRuleName] = useState('');
   237→  const [newRuleContent, setNewRuleContent] = useState('');
   238→  const [showAddRule, setShowAddRule] = useState(false);
   239→
   240→  const [appearance, setAppearance] = useState<AppearanceSettings>({
   241→    theme: 'dark',
   242→    fontFamily: 'Geist',
   243→    fontSize: 14,
   244→    tabSize: 2,
   245→    wordWrap: true,
   246→    lineNumbers: true,
   247→    minimap: true,

   250→
   251→  /* ── Persona handlers ── */
   252→
   253→  const handleCreatePersona = useCallback(() => {
   254→    if (!newPersonaName.trim() || !newPersonaPrompt.trim()) return;
   255→    const newPersona: Persona = {
   256→      id: `custom-${Date.now()}`,
   257→      name: newPersonaName.trim(),
   258→      icon: <Sparkles className="h-5 w-5" />,
   259→      description: newPersonaPrompt.trim().slice(0, 80),
   260→      systemPrompt: newPersonaPrompt.trim(),
   261→      focus: ['custom'],
   262→      color: 'text-orange-400',
   263→    };
   264→    setPersonas((prev) => [...prev, newPersona]);
   265→    setNewPersonaName('');
   266→    setNewPersonaPrompt('');
   267→  }, [newPersonaName, newPersonaPrompt]);
   268→
   269→  /* ── Rule handlers ── */
   270→
   271→  const handleAddRule = useCallback(() => {
   272→    if (!newRuleName.trim() || !newRuleContent.trim()) return;
   273→    const newRule: Rule = {
   274→      id: `rule-${Date.now()}`,
   275→      name: newRuleName.trim(),
   276→      content: newRuleContent.trim(),
   277→      enabled: true,
   278→    };
   279→    setRules((prev) => [...prev, newRule]);
   280→    setNewRuleName('');
   281→    setNewRuleContent('');
   282→    setShowAddRule(false);
   283→  }, [newRuleName, newRuleContent]);
   284→
   285→  const toggleRule = useCallback((id: string) => {
   286→    setRules((prev) => prev.map((r) => (r.id === id ? { ...r, enabled: !r.enabled } : r)));
   287→  }, []);
   288→
   289→  const deleteRule = useCallback((id: string) => {
   290→    setRules((prev) => prev.filter((r) => r.id !== id));
   291→  }, []);
   292→
   293→  /* ── Export / Import handlers ── */
   294→
   295→  const handleExport = useCallback(() => {
   296→    const config = { activePersona, personas, rules, appearance };
   297→    const blob = new Blob([JSON.stringify(config, null, 2)], { type: 'application/json' });
   298→    const url = URL.createObjectURL(blob);
   299→    const a = document.createElement('a');
   300→    a.href = url;
   301→    a.download = 'massive-number-config.json';
   302→    a.click();
   303→    URL.revokeObjectURL(url);
   304→  }, [activePersona, personas, rules, appearance]);
   305→
   306→  const handleImport = useCallback(() => {
   307→    const input = document.createElement('input');
   308→    input.type = 'file';
   309→    input.accept = '.json';
   310→    input.onchange = (e) => {
   311→      const file = (e.target as HTMLInputElement).files?.[0];
   312→      if (!file) return;
   313→      const reader = new FileReader();
   314→      reader.onload = (ev) => {
   315→        try {
   316→          const config = JSON.parse(ev.target?.result as string);
   317→          if (config.activePersona) setActivePersona(config.activePersona);
   318→          if (config.personas) setPersonas(config.personas);
   319→          if (config.rules) setRules(config.rules);
   320→          if (config.appearance) setAppearance((prev) => ({ ...prev, ...config.appearance }));
   321→        } catch {
   322→          // Invalid JSON — silently fail
   323→        }
   324→      };
   325→      reader.readAsText(file);
   326→    };
   327→    input.click();
   328→  }, []);
   329→
   330→  const handleReset = useCallback(() => {
   331→    setPersonas(PRESET_PERSONAS);
   332→    setRules(PRESET_RULES);
   333→    setActivePersona('builder');
   334→    setAppearance({
   335→      theme: 'dark',

   338→      tabSize: 2,
   339→      wordWrap: true,
   340→      lineNumbers: true,
   341→      minimap: true,
   342→      accentColor: 'emerald',
   343→    });

   367→              <motion.div
   368→                key={persona.id}

   370→                initial={{ opacity: 0, scale: 0.97 }}
   371→                animate={{ opacity: 1, scale: 1 }}
   372→                transition={{ duration: 0.2 }}

   374→                <Card

   382→                  {isActive && (

   384→                      layoutId="persona-glow"

   404→                            <Badge className="text-[9px] h-4 px-1.5 bg-emerald-500/20 text-emerald-400 border-emerald-500/30 hover:bg-emerald-500/30">
   405→                              Active
