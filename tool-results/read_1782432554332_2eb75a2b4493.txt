   109→function CodeBlock({
   110→  className,
   111→  children,
   112→  ...props
   113→}: ComponentPropsWithoutRef<'code'> & { children?: React.ReactNode }) {
   114→  const [copied, setCopied] = useState(false);
   115→  const match = /language-(\w+)/.exec(className || '');
   116→  const language = match ? match[1] : '';
   117→  const codeStr = String(children).replace(/\n$/, '');
   118→
   119→  const handleCopy = async () => {
   120→    await navigator.clipboard.writeText(codeStr);
   121→    setCopied(true);
   122→    setTimeout(() => setCopied(false), 2000);
   123→  };
   124→
   125→  if (match) {
   126→    return (
   127→      <div className="group relative my-3 rounded-lg overflow-hidden border border-border/50">
   128→        <div className="flex items-center justify-between bg-muted/60 px-3 py-1.5 text-xs text-muted-foreground">
   129→          <span className="font-mono">{language}</span>
   130→          <button
   131→            onClick={handleCopy}
   132→            className="flex items-center gap-1 rounded px-1.5 py-0.5 hover:bg-muted transition-colors"
   133→            aria-label="Copy code"
   134→          >
   135→            {copied ? (
   136→              <Check className="h-3 w-3 text-emerald-400" />
   137→            ) : (
   138→              <Copy className="h-3 w-3" />
   139→            )}
   140→            <span>{copied ? 'Copied' : 'Copy'}</span>
   141→          </button>
   142→        </div>
   143→        <SyntaxHighlighter
   144→          style={oneDark}
   145→          language={language}
   146→          PreTag="div"
   147→          customStyle={{
   148→            margin: 0,
   149→            padding: '0.75rem',
   150→            background: 'rgb(30, 30, 30)',
   151→            fontSize: '0.8125rem',
   152→            lineHeight: '1.5',
   153→          }}
   154→        >
   155→          {codeStr}
   156→        </SyntaxHighlighter>
   157→      </div>
   158→    );
   159→  }
   160→
   161→  return (
   162→    <code
   163→      className="rounded bg-muted px-1.5 py-0.5 text-sm font-mono"
   164→      {...props}
   165→    >
   166→      {children}
   167→    </code>
   168→  );
   169→}

   408→/* ───────── Main Chat Panel ───────── */
   409→
   410→export function ChatPanel() {
   411→  const [input, setInput] = useState('');
   412→  const [webGrounding, setWebGrounding] = useState(false);
   413→  const [isListening, setIsListening] = useState(false);
   414→  const [activePersona, setActivePersona] = useState<string>('default');
   415→  const scrollRef = useRef<HTMLDivElement>(null);
   416→  const textareaRef = useRef<HTMLTextAreaElement>(null);
   417→  const streamingMsgIdRef = useRef<string | null>(null);
   418→  const abortControllerRef = useRef<AbortController | null>(null);
   419→
   420→  const { chats, activeChatId, isStreaming, createChat, addMessage, updateMessage, setStreaming, clearChats } =
   421→    useChatStore();
   422→  const { selectedModel } = useModelStore();
   423→  const { addToast } = useUIStore();
   424→
   425→  const activeChat = chats.find((c) => c.id === activeChatId);
   426→  const tokenCount = estimateTokens(input);
   427→  const currentPersona = PERSONAS.find((p) => p.id === activePersona) || PERSONAS[0];
   428→
   429→  // Auto-scroll
   430→  useEffect(() => {
   431→    if (scrollRef.current) {
   432→      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
   433→    }
   434→  }, [activeChat?.messages]);
   435→
   436→  // Auto-resize textarea
   437→  useEffect(() => {
   438→    if (textareaRef.current) {
   439→      textareaRef.current.style.height = 'auto';
   440→      textareaRef.current.style.height =
   441→        Math.min(textareaRef.current.scrollHeight, 200) + 'px';
   442→    }
   443→  }, [input]);
   444→
   445→  // Simulated streaming: reveal text character by character
   446→  const simulateStreaming = useCallback(async (chatId: string, msgId: string, fullContent: string, model: string, tokens: number, cost: number, duration: number) => {
   447→    const chunkSize = 3;
   448→    let revealed = '';
   449→    for (let i = 0; i < fullContent.length; i += chunkSize) {
   450→      if (abortControllerRef.current?.signal.aborted) break;
   451→      revealed += fullContent.slice(i, i + chunkSize);
   452→      updateMessage(chatId, msgId, { content: revealed });
   453→      await new Promise(r => setTimeout(r, 12));
   454→    }
   455→    // Final update with metadata
   456→    updateMessage(chatId, msgId, {
   457→      content: fullContent,
   458→      model,
   459→      tokens,
   460→      cost,
   461→      duration,
   462→      isStreaming: false,
   463→    });
   464→    setStreaming(false);
   465→    streamingMsgIdRef.current = null;
   466→  }, [updateMessage, setStreaming]);
   467→
   468→  const sendMessage = useCallback(async (text: string, personaPrefix?: string) => {
   469→    if (!text.trim() || isStreaming) return;
   470→
   471→    let chatId = activeChatId;
   472→    if (!chatId) {
   473→      chatId = createChat(selectedModel, 'chat');
   474→    }
   475→
   476→    // Add user message
   477→    const userMsg: Message = {
   478→      id: `msg-${Date.now()}-user`,
   479→      role: 'user',
   480→      content: text,
   481→      timestamp: new Date(),
   482→    };
   483→    addMessage(chatId, userMsg);
   484→
   485→    // Create placeholder assistant message
   486→    const assistantMsgId = `msg-${Date.now()}-assistant`;
   487→    const assistantMsg: Message = {
   488→      id: assistantMsgId,
   489→      role: 'assistant',
   490→      content: '',
   491→      model: selectedModel,
   492→      isStreaming: true,
   493→      timestamp: new Date(),
   494→    };
   495→    addMessage(chatId, assistantMsg);
   496→    streamingMsgIdRef.current = assistantMsgId;
   497→    setStreaming(true);
   498→
   499→    // Create abort controller for this request
   500→    const controller = new AbortController();
   501→    abortControllerRef.current = controller;
   502→
   503→    try {
   504→      // Call the REST API
   505→      const allMessages = [...(activeChat?.messages || []), userMsg].map((m) => ({
   506→        role: m.role,
   507→        content: m.content,
   508→      }));
   509→
   510→      // Add persona prefix as system message if present
   511→      const systemPrefix = personaPrefix || currentPersona.prefix;
   512→      if (systemPrefix) {
   513→        allMessages.unshift({ role: 'system', content: systemPrefix });
   514→      }
   515→
   516→      const res = await fetch('/api/chat', {
   517→        method: 'POST',
   518→        headers: { 'Content-Type': 'application/json' },
   519→        body: JSON.stringify({
   520→          messages: allMessages,
   521→          model: selectedModel,
   522→          mode: 'chat',
   523→        }),
   524→        signal: controller.signal,
   525→      });
   526→
   527→      if (!res.ok) {
   528→        const errData = await res.json().catch(() => ({ error: 'Unknown error' }));
   529→        throw new Error(errData.error || `HTTP ${res.status}`);
   530→      }
   531→
   532→      const data = await res.json();
   533→
   534→      // Simulate streaming with the full response
   535→      const content = data.content || data.message?.content || 'No response received.';
   536→      const model = data.model || selectedModel;
   537→      const tokens = data.tokens || data.usage?.total_tokens || Math.ceil(content.length / 4);
   538→      const cost = data.cost || (tokens * 0.003);
   539→      const duration = data.duration || 0;
   540→
   541→      await simulateStreaming(chatId, assistantMsgId, content, model, tokens, cost, duration);
   542→    } catch (err: unknown) {
   543→      if (controller.signal.aborted) return;
   544→      const errorMessage = err instanceof Error ? err.message : 'Unknown error';
   545→      
   546→      // Fallback: show error message but also try socket.io
   547→      updateMessage(chatId, assistantMsgId, {
   548→        content: `I encountered an issue connecting to the AI service (${errorMessage}). Let me try an alternative approach...\n\n**Note:** Make sure your API keys are configured in Settings → Providers. MASSIVE NUMBER supports OpenAI, Anthropic, Google, DeepSeek, Meta, and more.`,
   549→        model: selectedModel,
   550→        tokens: 0,
   551→        cost: 0,
   552→        duration: 0,
   553→        isStreaming: false,
   554→      });
   555→      setStreaming(false);
   556→      streamingMsgIdRef.current = null;
   557→      addToast('Chat: Using fallback mode. Configure API keys in Settings.', 'info');
   558→    }
   559→  }, [
   560→    isStreaming,
   561→    activeChatId,
   562→    selectedModel,
   563→    activeChat,
   564→    currentPersona,
   565→    createChat,
   566→    addMessage,
   567→    updateMessage,
   568→    setStreaming,
   569→    simulateStreaming,
   570→    addToast,
   571→  ]);
   572→
   573→  const handleSend = useCallback(async () => {
   574→    const text = input.trim();
   575→    if (!text) return;
   576→    setInput('');
   577→    if (textareaRef.current) {
   578→      textareaRef.current.style.height = 'auto';
   579→    }
   580→    await sendMessage(text);
   581→  }, [input, sendMessage]);
   582→
   583→  const handleKeyDown = (e: React.KeyboardEvent) => {
   584→    if (e.key === 'Enter' && !e.shiftKey) {
   585→      e.preventDefault();
   586→      handleSend();
   587→    }
   588→  };
   589→
   590→  const handleSuggestionClick = (text: string) => {
   591→    setInput(text);
   592→    setTimeout(() => textareaRef.current?.focus(), 50);
   593→  };
   594→
   595→  const handleRegenerate = useCallback(async () => {
   596→    if (!activeChat || isStreaming) return;
   597→    const messages = activeChat.messages;
   598→    // Find the last user message
   599→    const lastUserMsg = [...messages].reverse().find((m) => m.role === 'user');
   600→    if (!lastUserMsg) return;
   601→    await sendMessage(lastUserMsg.content, currentPersona.prefix);
   602→  }, [activeChat, isStreaming, sendMessage, currentPersona]);
   603→
   604→  /* ───────── Export Functions ───────── */
   605→
   606→  const exportAsMarkdown = useCallback(() => {
   607→    if (!activeChat) return;
   608→    const lines: string[] = [`# ${activeChat.title}\n`];
   609→    activeChat.messages.forEach((msg) => {
   610→      if (msg.role === 'user') {
   611→        lines.push(`## You\n\n${msg.content}\n`);
   612→      } else if (msg.role === 'assistant') {
   613→        lines.push(`## MASSIVE NUMBER${msg.model ? ` (${msg.model})` : ''}\n\n${msg.content}\n`);
   614→        if (msg.tokens) lines.push(`*${msg.tokens} tokens | ${(msg.duration || 0) / 1000}s*\n`);
   615→      }
   616→    });
   617→    const blob = new Blob([lines.join('\n')], { type: 'text/markdown' });
   618→    const url = URL.createObjectURL(blob);
   619→    const a = document.createElement('a');
   620→    a.href = url;
   621→    a.download = `${activeChat.title.replace(/\s+/g, '_')}.md`;
   622→    a.click();
   623→    URL.revokeObjectURL(url);
   624→    addToast('Chat exported as Markdown', 'success');
   625→  }, [activeChat, addToast]);
   626→
   627→  const exportAsJSON = useCallback(() => {
   628→    if (!activeChat) return;
   629→    const data = {
   630→      title: activeChat.title,
   631→      model: activeChat.model,
   632→      mode: activeChat.mode,
   633→      exportedAt: new Date().toISOString(),
   634→      messages: activeChat.messages.map((m) => ({
   635→        role: m.role,
   636→        content: m.content,
   637→        model: m.model,
   638→        tokens: m.tokens,
   639→        duration: m.duration,
   640→        timestamp: m.timestamp,
   641→      })),
   642→    };
   643→    const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
   644→    const url = URL.createObjectURL(blob);
   645→    const a = document.createElement('a');
   646→    a.href = url;
   647→    a.download = `${activeChat.title.replace(/\s+/g, '_')}.json`;
   648→    a.click();
   649→    URL.revokeObjectURL(url);
   650→    addToast('Chat exported as JSON', 'success');
   651→  }, [activeChat, addToast]);
   652→
   653→  const handleClearChat = useCallback(() => {
   654→    clearChats();
   655→    addToast('Chat cleared', 'info');
   656→  }, [clearChats, addToast]);
   657→
   658→  // Find last assistant message for regenerate button
   659→  const lastAssistantMsgId = activeChat
   660→    ? [...activeChat.messages].reverse().find((m) => m.role === 'assistant' && !m.isStreaming)?.id
   661→    : null;
   662→
   663→  return (
   664→    <div className="flex flex-col h-full bg-background">
   665→      {/* Chat header with export menu */}
   666→      {activeChat && activeChat.messages.length > 0 && (
   667→        <div className="shrink-0 flex items-center justify-between px-4 py-2 border-b border-border/30 bg-card/30">
   668→          <div className="flex items-center gap-2">
   669→            <MessageSquare className="h-3.5 w-3.5 text-muted-foreground" />
   670→            <span className="text-xs text-muted-foreground truncate max-w-[200px]">
   671→              {activeChat.title}
   672→            </span>
   673→            <Badge variant="secondary" className="text-[8px] h-4 px-1.5 font-mono">
   674→              {activeChat.messages.length} msgs
   675→            </Badge>
   676→          </div>
   677→          <DropdownMenu>
   678→            <DropdownMenuTrigger asChild>
   679→              <Button variant="ghost" size="icon" className="h-7 w-7 text-muted-foreground hover:text-foreground">
   680→                <MoreVertical className="h-4 w-4" />
   681→              </Button>
   682→            </DropdownMenuTrigger>
   683→            <DropdownMenuContent align="end" className="w-48">
   684→              <DropdownMenuItem onClick={exportAsMarkdown} className="text-xs">
   685→                <FileText className="h-3.5 w-3.5 mr-2" />
   686→                Export as Markdown
   687→              </DropdownMenuItem>
   688→              <DropdownMenuItem onClick={exportAsJSON} className="text-xs">
   689→                <FileJson className="h-3.5 w-3.5 mr-2" />
   690→                Export as JSON
   691→              </DropdownMenuItem>
   692→              <DropdownMenuSeparator />
   693→              <DropdownMenuItem onClick={handleClearChat} className="text-xs text-red-400 focus:text-red-400">
   694→                <Trash2 className="h-3.5 w-3.5 mr-2" />
   695→                Clear Chat
   696→              </DropdownMenuItem>
   697→            </DropdownMenuContent>
   698→          </DropdownMenu>
   699→        </div>
   700→      )}
   701→
   702→      {/* Messages area */}
   703→      {activeChat && activeChat.messages.length > 0 ? (
   704→        <ScrollArea className="flex-1">
   705→          <div ref={scrollRef} className="max-w-3xl mx-auto py-4">
   706→            <AnimatePresence initial={false}>
   707→              {activeChat.messages.map((msg) => (
   708→                <ChatMessage
   709→                  key={msg.id}
   710→                  message={msg}
   711→                  isStreaming={msg.isStreaming}
   712→                  isLastAssistant={msg.id === lastAssistantMsgId}
   713→                  onRegenerate={msg.id === lastAssistantMsgId ? handleRegenerate : undefined}
   714→                />
   715→              ))}
   716→            </AnimatePresence>
   717→          </div>
   718→        </ScrollArea>
   719→      ) : (
   720→        <div className="flex-1 overflow-auto">
   721→          <EmptyState onSuggestionClick={handleSuggestionClick} />
   722→        </div>
   723→      )}
   724→
   725→      {/* Input area */}
   726→      <div className="border-t border-border/50 bg-background/95 backdrop-blur-sm">
   727→        <div className="max-w-3xl mx-auto px-4 py-3">
   728→          {/* Top row: model selector + persona + toggles */}
   729→          <div className="flex items-center gap-1 mb-2">
   730→            <ModelSelector />
   731→
   732→            {/* Persona Selector */}
   733→            <Popover>
   734→              <PopoverTrigger asChild>
   735→                <Button
   736→                  variant="ghost"
   737→                  size="sm"
   738→                  className={`h-8 text-[10px] px-2 gap-1 ${
   739→                    activePersona !== 'default'
   740→                      ? 'text-amber-400 bg-amber-400/10'
   741→                      : 'text-muted-foreground'
   742→                  }`}
   743→                >
   744→                  {currentPersona.icon}
   745→                  <span className="hidden sm:inline">{currentPersona.name}</span>
   746→                  <ChevronDown className="h-2.5 w-2.5" />
   747→                </Button>
   748→              </PopoverTrigger>
   749→              <PopoverContent className="w-64 p-2" align="start">
   750→                <div className="space-y-0.5">
   751→                  <p className="text-[10px] font-medium text-muted-foreground px-2 py-1 uppercase tracking-wider">
   752→                    AI Persona
   753→                  </p>
   754→                  {PERSONAS.map((persona) => (
   755→                    <button
   756→                      key={persona.id}
   757→                      onClick={() => setActivePersona(persona.id)}
   758→                      className={`w-full flex items-center gap-2.5 px-2 py-2 rounded-md text-left transition-colors ${
   759→                        activePersona === persona.id
   760→                          ? 'bg-emerald-500/10 text-emerald-400'
   761→                          : 'hover:bg-muted/50 text-foreground'
   762→                      }`}
   763→                    >
   764→                      <span className={`shrink-0 ${activePersona === persona.id ? 'text-emerald-400' : 'text-muted-foreground'}`}>
   765→                        {persona.icon}
   766→                      </span>
   767→                      <div className="flex-1 min-w-0">
   768→                        <div className="text-xs font-medium">{persona.name}</div>
   769→                        <div className="text-[9px] text-muted-foreground truncate">{persona.description}</div>
   770→                      </div>
   771→                      {activePersona === persona.id && (
   772→                        <Check className="h-3.5 w-3.5 text-emerald-400 shrink-0" />
   773→                      )}
   774→                    </button>
   775→                  ))}
   776→                </div>
   777→              </PopoverContent>
   778→            </Popover>
   779→
   780→            <div className="flex-1" />
   781→            <TooltipProvider delayDuration={200}>
   782→              <Tooltip>
   783→                <TooltipTrigger asChild>
   784→                  <Button
   785→                    variant="ghost"
   786→                    size="icon"
   787→                    className={`h-8 w-8 ${
   788→                      webGrounding ? 'text-emerald-400 bg-emerald-400/10' : 'text-muted-foreground'
   789→                    }`}
   790→                    onClick={() => setWebGrounding(!webGrounding)}
   791→                  >
   792→                    <Globe className="h-4 w-4" />
   793→                  </Button>
   794→                </TooltipTrigger>
   795→                <TooltipContent>Web Grounding {webGrounding ? 'ON' : 'OFF'}</TooltipContent>
   796→              </Tooltip>
   797→              <Tooltip>
   798→                <TooltipTrigger asChild>
   799→                  <Button variant="ghost" size="icon" className="h-8 w-8 text-muted-foreground">
   800→                    <Paperclip className="h-4 w-4" />
   801→                  </Button>
   802→                </TooltipTrigger>
   803→                <TooltipContent>Attach file</TooltipContent>
   804→              </Tooltip>
   805→              <Tooltip>
   806→                <TooltipTrigger asChild>
   807→                  <Button
   808→                    variant="ghost"
   809→                    size="icon"
   810→                    className={`h-8 w-8 ${
   811→                      isListening ? 'text-red-400 bg-red-400/10' : 'text-muted-foreground'
   812→                    }`}
   813→                    onClick={() => setIsListening(!isListening)}
   814→                  >
   815→                    <Mic className="h-4 w-4" />
   816→                  </Button>
   817→                </TooltipTrigger>
   818→                <TooltipContent>Voice input {isListening ? '(listening...)' : ''}</TooltipContent>
   819→              </Tooltip>
   820→            </TooltipProvider>
   821→          </div>
   822→
   823→          {/* Input row */}
   824→          <div className="flex items-end gap-2">
   825→            <Textarea
   826→              ref={textareaRef}
   827→              value={input}
   828→              onChange={(e) => setInput(e.target.value)}
   829→              onKeyDown={handleKeyDown}
   830→              placeholder="Ask anything... (Shift+Enter for new line)"
   831→              className="min-h-[40px] max-h-[200px] resize-none bg-card border-border/50 focus-visible:ring-emerald-500/30 placeholder:text-muted-foreground/60"
   832→              rows={1}
   833→              disabled={isStreaming}
   834→            />
   835→            <Button
   836→              onClick={handleSend}
   837→              disabled={!input.trim() || isStreaming}
   838→              size="icon"
   839→              className="h-10 w-10 shrink-0 bg-emerald-600 hover:bg-emerald-700 text-white disabled:opacity-40 transition-all"
   840→            >
   841→              {isStreaming ? (
   842→                <Loader2 className="h-4 w-4 animate-spin" />
   843→              ) : (
   844→                <Send className="h-4 w-4" />
   845→              )}
   846→            </Button>
   847→          </div>
   848→
   849→          {/* Bottom hint with token counter */}
   850→          <div className="flex items-center justify-between mt-1.5">
   851→            <span className="text-[10px] text-muted-foreground/50">
   852→              <MessageSquare className="h-3 w-3 inline mr-1" />
   853→              Powered by MASSIVE NUMBER · Multi-model orchestration
   854→            </span>
   855→            {input.length > 0 && (
   856→              <span className="text-[10px] text-muted-foreground/50 tabular-nums">
   857→                ≈{tokenCount} tokens
   858→              </span>
   859→            )}
   860→          </div>
   861→        </div>
   862→      </div>
   863→    </div>
   864→  );
   865→}
